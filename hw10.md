## 2.

```C
#include<stdio.h>
void insert(int *, int, int *);
int main(void) {
	int arr[6] = {60,65,75,80,90,95};
	int new_arr[7];
	int x;
	scanf("%d", &x);
	insert(arr,x,new_arr);
	for (int i = 0; i < 7; ++i) {
		printf("%d ", new_arr[i]);
	}
	
} 
void insert(int arr1[], int insert_number, int arr2[]) {//在arr1中插入一个数，并储存在arr2中 
	for (int i = 0; i < 6; ++i) 
		arr2[i] = arr1[i];
	arr2[6] = insert_number;
	int temp = arr2[6];
	
	for (int i = 0; i < 6; ++i) {
		if (arr2[6] < arr2[i]) {
			
			for (int j = 5; j >= i; --j) 
				
				arr2[j + 1] = arr2[j];
			arr2[i] = temp;
			
			break;
		}
			
	}
}
```
* 选择插在头、中间、尾巴的三个测试数据可以检验算法的全面性，即保证可以在任意位置插入一个数。
